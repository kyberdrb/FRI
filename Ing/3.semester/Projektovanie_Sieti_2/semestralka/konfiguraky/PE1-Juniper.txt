eBGP

configure
set interfaces fe-0/0/0.0 family inet address 10.42.12.1/24
set interfaces lo0.0 family inet address 1.1.1.1/32
commit and-quit
show configuration interfaces

#Nastavime eBGP peering

configure
set routing-options autonomous-system 65001
set protocols bgp group EBGP neighbor 10.42.12.2 peer-as 65002
commit and-quit
show configuration routing-options
show configuration protocols bgp
show bgp neighbor 10.42.12.2 | match Established

#Ohlasime prefixy eBGP peerovi

configure
edit policy-options policy-statement EBGP_ADVERTISE_LOOPBACK
set term ADVERTISE_LO0 from protocol direct
set term ADVERTISE_LO0 from route-filter 1.1.1.1/32 exact
set term ADVERTISE_LO0 then accept
top
show configuration policy-options
configure
set protocols bgp group EBGP neighbor 10.42.12.2 export EBGP_ADVERTISE_LOOPBACK
commit and-quit

#Overenie BGP procesu

show configuration protocols bgp
show bgp summary
show route receive-protocol bgp 10.42.12.2
show route protocol bgp
